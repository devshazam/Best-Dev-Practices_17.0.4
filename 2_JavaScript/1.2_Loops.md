# for для всех
    > for(let i = 0; i < 10; i++){...}
    	> for(let i = 0, j = 10; i < 10 && j > 0; i++, j--){...}


# for of - по значению // для массивов и строк
	> for(let i of obj){...}

# for in - по индексу// для массивов и объектов и строк
	⛔index имеет тип string
	> for(let i in obj){...}

# WHILE - до тех пор пока утверждение в скобках = true 
    > while (x < 10){ ... }
